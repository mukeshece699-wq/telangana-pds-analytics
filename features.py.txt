import numpy as np

def create_features(df):
    # Utilization Ratio
    df['utilization_ratio'] = df['noOfTrans'] / (df['totalRcs'] + 1)

    # Commodity ratio
    if 'rice' in df.columns and 'wheat' in df.columns:
        df['rice_wheat_ratio'] = df['rice'] / (df['wheat'] + 1)

    # volatility (group by shop)
    df['trans_std'] = df.groupby('shopNo')['noOfTrans'].transform('std')

    df = df.fillna(0)

    return df
